CREATE PROCEDURE Silver.LoadDimensionsFromBronze
AS
BEGIN
    -- Insert into DimDateTime from Case_Date
    INSERT INTO silver.DimDateTime (FullDate)
    SELECT  CAST(Case_Date AS DATE) AS FullDate
    FROM bronze.Synthetic_HumanTraffickingCases b
    LEFT JOIN silver.DimDateTime d ON d.FullDate = CAST(b.Case_Date AS DATE)
    WHERE d.DateID IS NULL;

    -- Insert into DimVictim
    INSERT INTO silver.DimVictim (Name, Age, Gender, Nationality)
    SELECT  
        b.Victim_Name,
        b.Victim_Age,
        b.Victim_Gender,
        b.Victim_Nationality
    FROM bronze.Synthetic_HumanTraffickingCases b
    LEFT JOIN silver.DimVictim v ON 
        v.Name = b.Victim_Name AND
        v.Age = b.Victim_Age AND
        v.Gender = b.Victim_Gender AND
        v.Nationality = b.Victim_Nationality
    WHERE v.VictimID IS NULL;

    -- Insert into DimTrafficker
    INSERT INTO silver.DimTrafficker (Name, Gender, Nationality)
    SELECT  
        b.Trafficker_Name,
        b.Trafficker_Gender,
        b.Trafficker_Nationality
    FROM bronze.Synthetic_HumanTraffickingCases b
    LEFT JOIN silver.DimTrafficker t ON 
        t.Name = b.Trafficker_Name AND
        t.Gender = b.Trafficker_Gender AND
        t.Nationality = b.Trafficker_Nationality
    WHERE t.TraffickerID IS NULL;

    -- Insert into DimTransportationType
    INSERT INTO silver.DimTransportationType (TransportationMode)
    SELECT  b.TransportationMode
    FROM bronze.Synthetic_HumanTraffickingCases b
    LEFT JOIN silver.DimTransportationType t ON t.TransportationMode = b.TransportationMode
    WHERE t.TransportationTypeID IS NULL;

    -- Insert into DimRecruitmentType
    INSERT INTO silver.DimRecruitmentType (RecruitmentTypeName)
    SELECT  b.Recruitment_Method
    FROM bronze.Synthetic_HumanTraffickingCases b
    LEFT JOIN silver.DimRecruitmentType r ON r.RecruitmentTypeName = b.Recruitment_Method
    WHERE r.RecruitmentTypeID IS NULL;

    -- Insert into DimExploitationType
    INSERT INTO silver.DimExploitationType (ExploitationType)
    SELECT  b.Exploitation_Type
    FROM bronze.Synthetic_HumanTraffickingCases b
    LEFT JOIN silver.DimExploitationType e ON e.ExploitationType = b.Exploitation_Type
    WHERE e.ExploitationTypeID IS NULL;

    -- Insert into DimOperation
    INSERT INTO silver.DimOperation (InterventionType)
    SELECT  b.Intervention_Type
    FROM bronze.Synthetic_HumanTraffickingCases b
    LEFT JOIN silver.DimOperation o ON o.InterventionType = b.Intervention_Type
    WHERE o.OperationID IS NULL;

    -- Insert into DimLocation (Location of Exploitation)
    INSERT INTO silver.DimLocation (Address, Country, Urban_Rural)
    SELECT  
        b.Location_of_Exploitation,
        b.Exploitation_Country,
        b.Location_of_Exploitation AS Urban_Rural
    FROM bronze.Synthetic_HumanTraffickingCases b
    LEFT JOIN silver.DimLocation l ON 
        l.Address = b.Location_of_Exploitation AND
        l.Country = b.Exploitation_Country
    WHERE l.LocationID IS NULL;

END;
GO

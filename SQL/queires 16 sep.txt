SELECT  
-- count(*)
distinct v.VictimID, d.DateID
FROM  Bronze.Synthetic_HumanTraffickingCases b

INSERT INTO Silver.mapVictimLocationID (VictimID, LocationID)
SELECT 
count(*)
distinct v.VictimID, l.LocationID
FROM  Bronze.Synthetic_HumanTraffickingCases b

JOIN Silver.DimLocation l ON b.Address = l.Address AND b.Zipcode = l.Zipcode;

INSERT INTO Silver.mapVictimTraffickerID (VictimID, TraffickerID)
SELECT 
v.VictimID, t.TraffickerID
FROM Bronze.Synthetic_HumanTraffickingCases b

JOIN Silver.DimTrafficker t ON b.TraffickerNationalID = t.NationalID and t.IsCurrent = 1;

INSERT INTO Silver.mapVictimRecruitmenttypeID (VictimID, RecruitmentTypeID)
SELECT  
DISTINCT v.VictimID, r.RecruitmentTypeID
FROM Bronze.Synthetic_HumanTraffickingCases b
JOIN Silver.DimVictim v ON b.VictimNationalID = v.NationalID and IsCurrent = 1
JOIN  Silver.DimRecruitmentType r ON b.RecruitmentTypeName = r.RecruitmentTypeName
and b.RecruitmentDescription = r.description;

INSERT INTO Silver.FactIncident (
    DateID,
    LocationID,
    VictimID,
    TraffickerID,
    RecruitmentTypeID,
    TransportTypeID,
    ExploitationTypeID,
    CaseID
)

SELECT 
    COUNT(*) 
    -- If you want distinct combinations instead:
    -- COUNT(DISTINCT v.VictimID, vd.DateID, vl.LocationID, vt.TraffickerID,
    --                      vr.RecruitmentTypeID, vtt.TransportTypeID, 
    --                      ve.ExploitationTypeID, vc.CaseID) 
FROM Silver.DimVictim v

-- Victim–Case (anchor)
JOIN Silver.mapVictimCaseID vc
    ON v.VictimID = vc.VictimID

-- Victim–Date
JOIN Silver.mapVictimDateID vd 
    ON v.VictimID = vd.VictimID
   AND vc.CaseID  = vd.CaseID   -- case alignment

-- Victim–Location
JOIN Silver.mapVictimLocationID vl 
    ON v.VictimID = vl.VictimID
   AND vc.CaseID  = vl.CaseID

-- Victim–Trafficker
JOIN Silver.mapVictimTraffickerID vt 
    ON v.VictimID = vt.VictimID
   AND vc.CaseID  = vt.CaseID

-- Victim–Recruitment
JOIN Silver.mapVictimRecruitmentTypeID vr 
    ON v.VictimID = vr.VictimID
   AND vc.CaseID  = vr.CaseID

-- Victim–Transport
JOIN Silver.mapVictimTransportTypeID vtt 
    ON v.VictimID = vtt.VictimID
   AND vc.CaseID  = vtt.CaseID

-- Victim–Exploitation
JOIN Silver.mapVictimExploitationTypeID ve 
    ON v.VictimID = ve.VictimID
   AND vc.CaseID  = ve.CaseID;


SELECT 
-- count(*)
DISTINCT
    vd.DateID,
    vl.LocationID,
    v.VictimID,
    vt.TraffickerID,
    vr.RecruitmentTypeID,
    vtt.TransportTypeID,
    ve.ExploitationTypeID,
    vc.CaseID
FROM Silver.DimVictim v

-- Victim–Date
JOIN Silver.mapVictimDateID vd 
    ON v.VictimID = vd.VictimID

-- Victim–Location
JOIN Silver.mapVictimLocationID vl 
    ON v.VictimID = vl.VictimID

-- Victim–Trafficker
JOIN Silver.mapVictimTraffickerID vt 
    ON v.VictimID = vt.VictimID

-- Victim–Recruitment
JOIN Silver.mapVictimRecruitmentTypeID vr 
    ON v.VictimID = vr.VictimID

-- Victim–Transport
JOIN Silver.mapVictimTransportTypeID vtt 
    ON v.VictimID = vtt.VictimID

-- Victim–Exploitation
JOIN Silver.mapVictimExploitationTypeID ve 
    ON v.VictimID = ve.VictimID

-- Victim–Case
JOIN Silver.mapVictimCaseID vc
    ON v.VictimID = vc.VictimID;
